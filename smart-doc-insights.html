When user asks a question: embed the question → search ChromaDB for top 5 most similar chunks → inject those chunks into a prompt → send to GPT-3.5-turbo → return answer with page citations.

The RAG Query Function
# query.py — the heart of your system
from langchain.vectorstores import Chroma
from langchain.embeddings import HuggingFaceEmbeddings
import openai

embeddings = HuggingFaceEmbeddings(model_name="all-MiniLM-L6-v2")
vectordb   = Chroma(persist_directory="./chroma_db", embedding_function=embeddings)

def ask_document(question: str, doc_id: str):
    # 1. Find the most relevant chunks
    results = vectordb.similarity_search(
        question, k=5,
        filter={"doc_id": doc_id}
    )

    # 2. Build context from chunks
    context = "\n\n".join([r.page_content for r in results])

    # 3. Send to GPT with the context
    response = openai.chat.completions.create(
        model="gpt-3.5-turbo",
        messages=[
            {"role": "system", "content": "Answer using ONLY the context provided. Always cite page numbers."},
            {"role": "user", "content": f"Context:\n{context}\n\nQuestion: {question}"}
        ]
    )

    # 4. Return answer + source metadata
    return {
        "answer": response.choices[0].message.content,
        "sources": [{"page": r.metadata["page"], "text": r.page_content[:200]} for r in results]
    }
